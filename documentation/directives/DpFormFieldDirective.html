از این دایرکتیو برای تغییر استایل mat-form-field های Angular material استفاده شده است. در این دایرکتیو کلاس 'dp-blue-form-field' به mat-form-field اضافه می شود. از این کلاس در فایل dp-form-field.scss برای تغییر استایل mat-form-field استفاده می شود.

import {Directive, ElementRef, OnInit, Renderer2} from '@angular/core';

@Directive({
    selector: '[appDpFormField]'
})
export class DpFormFieldDirective implements OnInit {
    private inputClass = 'dp-blue-form-field';

    constructor(private renderer: Renderer2, private el: ElementRef) {
    }

    ngOnInit(): void {
        this.renderer.addClass(this.el.nativeElement, this.inputClass);
    }


}
