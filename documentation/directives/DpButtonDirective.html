از این دایرکتیو برای تغییر استایل button ها استفاده شده است. ورودی appDpButton کلاس button را مشخص می کند. از این کلاس در فایل dp-button.scss برای استایل دهی button استفاده می شود. کلاس هایی که در این فایل برای آن ها استایل اضافه شده است عبارتند از :

  • blue
  • green
  • orange
  • dark-blue
  • blue-outline
  • orange-outline
  • red-outline
  • gray-outline

import {Directive, ElementRef, Input, OnInit, Renderer2} from '@angular/core';

@Directive({
    selector: '[appDpButton]'
})
export class DpButtonDirective implements OnInit {
    @Input() appDpButton = 'blue';

    constructor(private renderer: Renderer2, private el: ElementRef) {
    }

    ngOnInit(): void {
        this.renderer.addClass(this.el.nativeElement, this.appDpButton);
        this.renderer.addClass(this.el.nativeElement, 'dp-button');
    }
}
