import {Component, EventEmitter, Output} from '@angular/core';
import {animate, query, state, style, transition, trigger} from '@angular/animations';

@Component({
    selector: 'app-rate-star-bar',

export class RateStarBarComponent {

    @Output() select = new EventEmitter();
    selectedRate = -1;
    starCount: number[];
    unselectAnimationState = 'init';

    constructor() {
        this.starCount = Array(5).fill(-1).map((x, i) => (i + 1)); // [0,1,2,3,4]
    }

    setUnselectAnimationState(unselectState: string) {
        this.unselectAnimationState = unselectState;
    }

    onSelect(selection: number) {
        this.selectedRate = selection;
        this.select.emit(selection);
    }

    onDeselect() {
        this.selectedRate = -1;
        this.select.emit(undefined);
    }

}

<div id="rate-star-bar">
    <div class="unselected-star-container">
        <div (@rateSelectAnimation.start)="setUnselectAnimationState('start')"
             *ngFor="let starNumber of starCount"
             [@rateSelectAnimation]="{value: selectedRate!==-1?'hide':'show', params:{xMove: (3-starNumber)*42}}"
             class="star-container">
            <mat-icon
                    (click)="selectedRate===-1?onSelect(starNumber):null"
                    [svgIcon]="selectedRate>=starNumber?'svg_baseline_star':'svg_outline_star'"
                    class="star-icon"
            ></mat-icon>
            <span class="star-text">{{starNumber}}</span>
        </div>
    </div>
    <div
            *ngIf="selectedRate!==-1"
            [@rateUnselectAnimation]="unselectAnimationState"
            class="selected-star-container">
        <mat-icon
                (click)="onDeselect()"
                svgIcon="svg_baseline_star"
                class="selected-star-icon"
        ></mat-icon>
        <span (click)="onDeselect()" class="selected-star-text">{{selectedRate}}</span>
    </div>
</div>
